"""Cree un programa que lea nombres de canciones
de un archivo (línea por línea) y guarde en otro
archivo los mismos nombres ordenados
alfabéticamente."""
import shutil


def abrir_y_leer_archivo(semana_8):
    with open(semana_8) as file:
        for line in file.readlines():
            print(line)
            
abrir_y_leer_archivo('lista_de_canciones.txt')  


def leer_datos_guardarlos(semana_8):
    with open(semana_8) as file:
        for line in file:
            fichero = open("canciones_ordenadas.txt", 'w')
            fichero.write(line + "\n")
            
            
def convertir_y_ordernar(semana_8):
    lines = []
    with open(semana_8, 'r') as f:
        for line in f:
            lines.append(line.strip())
    lines.sort()
    with open("canciones_ordenadas", "w") as txt_file:
        for line in lines:   
            txt_file.write("".join(line)+ "\n") 
        
convertir_y_ordernar('lista_de_canciones.txt')
